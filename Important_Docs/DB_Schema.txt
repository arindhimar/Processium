==USERS & ROLES==
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    full_name VARCHAR(150) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE roles (
    id SERIAL PRIMARY KEY,
    role_name VARCHAR(100) UNIQUE NOT NULL
);

user_roles (Many-to-Many)
CREATE TABLE user_roles (
    id SERIAL PRIMARY KEY,
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    role_id INT REFERENCES roles(id) ON DELETE CASCADE
);

==REQUISITION MODULE==
CREATE TABLE requisitions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title VARCHAR(200) NOT NULL,
    department VARCHAR(150) NOT NULL,
    job_description TEXT NOT NULL,
    status VARCHAR(50) CHECK (status IN ('OPEN', 'IN_PROGRESS', 'CLOSED')) DEFAULT 'OPEN',
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    closed_at TIMESTAMP
);

CREATE INDEX idx_requisition_status ON requisitions(status);


==WORKFLOW TEMPLATE ENGINE==
CREATE TABLE workflow_templates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(150) NOT NULL,
    department VARCHAR(150),
    is_active BOOLEAN DEFAULT TRUE,
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE template_steps (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    template_id UUID REFERENCES workflow_templates(id) ON DELETE CASCADE,
    step_name VARCHAR(150) NOT NULL,
    step_order INT NOT NULL,
    is_mandatory BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_template_step_order ON template_steps(template_id, step_order);

==RACI CONFIGURATION==
CREATE TABLE step_raci_assignments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    template_step_id UUID REFERENCES template_steps(id) ON DELETE CASCADE,
    user_id UUID REFERENCES users(id),
    raci_type VARCHAR(20) CHECK (raci_type IN ('RESPONSIBLE','ACCOUNTABLE','CONSULTED','INFORMED'))
);


==HIRING INSTANCE (Execution Layer)==
CREATE TABLE hiring_instances (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    requisition_id UUID REFERENCES requisitions(id) ON DELETE CASCADE,
    template_id UUID REFERENCES workflow_templates(id),
    status VARCHAR(50) CHECK (status IN ('ACTIVE','COMPLETED','CANCELLED')) DEFAULT 'ACTIVE',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);



instance_steps (Runtime Copy of Template Steps)
CREATE TABLE instance_steps (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    hiring_instance_id UUID REFERENCES hiring_instances(id) ON DELETE CASCADE,
    template_step_id UUID REFERENCES template_steps(id),
    step_name VARCHAR(150),
    step_order INT,
    status VARCHAR(50) CHECK (status IN ('PENDING','IN_PROGRESS','COMPLETED')) DEFAULT 'PENDING',
    started_at TIMESTAMP,
    completed_at TIMESTAMP
);

CREATE INDEX idx_instance_status ON instance_steps(status);


==COMMENTS SYSTEM==
Supports:
Step-level comments
Global (instance-level) comments
Tagging

CREATE TABLE comments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    hiring_instance_id UUID REFERENCES hiring_instances(id) ON DELETE CASCADE,
    instance_step_id UUID REFERENCES instance_steps(id) ON DELETE SET NULL,
    user_id UUID REFERENCES users(id),
    comment_text TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

If:
instance_step_id IS NULL → Global comment
NOT NULL → Step-level comment

CREATE TABLE comment_tags (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    comment_id UUID REFERENCES comments(id) ON DELETE CASCADE,
    tagged_user_id UUID REFERENCES users(id)
);


==AUDIT LOGGING==
CREATE TABLE audit_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    entity_type VARCHAR(100) NOT NULL,
    entity_id UUID NOT NULL,
    action_type VARCHAR(100) NOT NULL,
    performed_by UUID REFERENCES users(id),
    old_value JSONB,
    new_value JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

REQUISITION_CREATED
STEP_STATUS_UPDATED
TEMPLATE_MODIFIED
